    // 反转后 n-p 个元素

    // 反转整个数组

//使用一个变量 k 记录不重复元素的位置，通过内层循环检查当前元素是否与前面的元素重复
void removeDuplicates(SeqList* list) {
    int k = 0;
    for (int i = 0; i < list->len; i++) {
        int j = 0;
        while (j < k && list->arr[i] != list->arr[j]) {
            j++;
        }
        if (j == k) {
            list->arr[k++] = list->arr[i];
        }
    }
    list->len = k;
}

int findM(SeqList s) {
    int c = -1, cnt = 0;

        if (cnt == 0) {

            cnt = 1;
        } else if (s.arr[i] == c) {

            cnt--;

            cnt++;
        }
    }

    if (cnt > s.len / 2)

}

int findMinPositive(int A[], int n) {
    // 动态分配一个大小为 n+2 的辅助数组，初始化为 0
    int *B = (int *)calloc(n + 2, sizeof(int)); 
    
    // 1. 计数标记
    for (int i = 0; i < n; i++) {
        // 只有在 1 到 n 范围内的数才需要记录，其余大数和负数直接无视
        if (A[i] > 0 && A[i] <= n) {
            B[A[i]] = 1;
        }
    }
    
    // 2. 扫描寻找第一个未出现的正整数
    for (int i = 1; i <= n + 1; i++) {
        if (B[i] == 0) {
            free(B);
            return i; // 找到了，直接返回
        }
    }
    
    free(B);
    return n + 1;
}

int findMinAbsent(int arr[], int n) {
    if (n == 0) return 1;

    int max = n + 1;
    char present[max]; // 模拟布尔数组

    for (int i = 0; i < max; ++i)
        present[i] = 0;

    for (int i = 0; i < n; ++i) {
        if (arr[i] > 0 && arr[i] <= n)
            present[arr[i]] = 1;
    }

    for (int i = 1; i <= n + 1; ++i)
        if (!present[i]) return i;

    return -1; // 不可达
}

## 数组强化12

**12、一个长度为 L 的升序序列 S，处在第 L/2（向上取整）个位置的数称为 S 的中位数。例如 S1=(11,13,15,17,19)，S1 的中位数为 15，两个序列的中位数是含他们所有元素升序序列的中位数。例如，若 S2=(2,4,6,8,20)，则 S1 和 S2 的中位数是 11。现在有两个等长的升序序列 A 和 B，试设计一个在时间和空间都尽可能高效的算法，找出两个序列 A 和 B 的中位数（2011 年统考题）**

不借助第三个数组，**用两个指针 i 和 j 遍历 A 和 B，合并查找第 len/2 个最小值**。这是一个模拟归并排序中 "取中位" 的过程，**不需要实际合并整个数组**。

```c
int findMedian(SeqList A, SeqList B) {
    int i = 0, j = 0, count = 0;
    int median = 0;
    int target = A.len;  // A 和 B 等长，总长为 2 * len，中位数在第 len 位（从 1 开始）

    while (count < target) {
        if (i < A.len && (j >= B.len || A.arr[i] < B.arr[j])) {//“如果还能从 A 中取元素，并且要么 B 已经没有元素了，要么 A 的当前元素更小”，那就从 A 取出当前元素。
            median = A.arr[i++];
        } else {
            median = B.arr[j++];
        }
        count++;
    }

    return median;
}

```

## 数组强化13 fangqi

编写代码，将自然数 1∼n^2 螺旋写入二维矩阵当中，如下图所示。

````
1   2   3   4
12  13  14  5
11  16  15  6
10  9   8   7
````



```c
#define maxsize 50

void func(int n, int a[][maxsize]) {
    int value = 1;                    // 当前要填入的数值
    int minCol = 0, maxCol = n - 1;   // 列的边界范围
    int minRow = 0, maxRow = n - 1;   // 行的边界范围

    while (value <= n * n) {
        // 1. 填充上边：从左到右
        for (int i = minCol; i <= maxCol; i++) {
            a[minRow][i] = value++;
        }
        minRow++;  // 上边填完，下一行开始

        // 2. 填充右边：从上到下
        for (int i = minRow; i <= maxRow; i++) {
            a[i][maxCol] = value++;
        }
        maxCol--;  // 右边填完，左一列开始

        // 3. 填充下边：从右到左
        for (int i = maxCol; i >= minCol; i--) {
            a[maxRow][i] = value++;
        }
        maxRow--;  // 下边填完，上一行开始

        // 4. 填充左边：从下到上
        for (int i = maxRow; i >= minRow; i--) {
            a[i][minCol] = value++;
        }
        minCol++;  // 左边填完，右一列开始
    }
}
```



## KMP last

KMP（Knuth-Morris-Pratt）算法是一种**字符串匹配算法**，它的基本设计思想是：

当模式串（要查找的字符串）与主串（被查找的字符串）在某一位置发生不匹配时，**不是简单地将模式串向后移动一位**，而是利用模式串自身已匹配部分的信息（即**“部分匹配表”或“next数组”**），确定模式串下一次开始匹配的位置，从而**避免对主串中已匹配过的字符进行不必要的重复比较**。

简而言之，就是**利用模式串的内部结构，在失配时尽可能大地向右滑动模式串，以提高匹配效率。**

```c
// 函数功能：计算模式串 pattern 的 KMP next 数组（优化前的标准形式）
// 参数说明：
//   pattern[]: 模式串（要匹配的字符串）
//   next[]:    输出的 next 数组，长度为 n
//   n:         模式串的长度
void get_next(char pattern[], int next[], int n) {
    // 初始化：next[0] = 0
    // 因为单个字符没有真前缀和真后缀，所以初始匹配长度为0
    next[0] = 0;

    // i：当前正在计算 next[i] 的位置（从1开始）
    // len：当前已匹配的前缀长度（也表示 pattern[0..len-1] 与 pattern[i-len..i-1] 匹配）
    int i = 1, len = 0;

    // 遍历模式串，计算每个位置的最长相等前后缀长度
    while (i < n) {
        // 情况1：当前字符 pattern[i] 与前缀下一个字符 pattern[len] 相等
        if (pattern[i] == pattern[len]) {
            // 匹配成功，len 增加1，next[i] = len，然后 i 向后移动
            next[i++] = ++len;
        } else {
            // 情况2：不相等，尝试回退
            // 注意：这里不是标准 KMP 的写法（标准写法应使用 while 循环回退 len = next[len-1]）
            // 而是简化处理：直接看是否能和 pattern[0] 匹配

            if (pattern[i] == pattern[0]) {
                // 如果当前字符等于 pattern[0]，则 next[i] = 1
                next[i++] = 1;
                len = 1;  // 当前匹配长度设为1
            } else {
                // 完全不匹配，next[i] = 0
                next[i++] = 0;
                len = 0;  // 匹配长度归零
            }
        }
    }

    // 将 next 数组整体右移一位（为 KMP 主算法做准备）
    // 例如：原 next[0..n-1] = [0,1,0,...] → 右移后变成 [-1,0,1,0,...]
    for (int j = n - 1; j > 0; --j) {
        next[j] = next[j - 1];
    }
    // next[0] 设为 -1，这是 KMP 算法中常见的哨兵值，
    // 表示当主串和模式串第一个字符就不匹配时，模式串指针跳到 -1，主串指针前进
    next[0] = -1;
}
```

```c
// KMP 主匹配函数：在 test（主串）中查找 pattern（模式串）
// 时间复杂度：O(m + n)，空间复杂度：O(n)（用于存储 next 数组）
void kmp(char pattern[], char test[]) {
    // 获取模式串和主串的长度
    int n = len(pattern);  // 模式串长度（通常应为 strlen(pattern)，此处假设 len 是自定义或宏）
    int m = len(test);     // 主串长度

    // 动态分配 next 数组，大小为模式串长度 n
    int *next = new int[n];

    // 调用 get_next 函数，预处理得到 pattern 的 next 数组
    get_next(pattern, next, n);

    // i：主串 test 的当前下标；j：模式串 pattern 的当前下标
    int i = 0, j = 0;

    // 遍历主串，直到结尾
    while (i < m) {
        // 匹配成功条件：j 已走到模式串最后一个字符（n-1），且当前字符也匹配
        if (j == n - 1 && test[i] == pattern[j]) {
            cout << "匹配成功！,位置为：";
            // 输出匹配起始位置（从0开始计数）
            cout << i - j << endl;
            return;  // 找到第一个匹配就返回（若要找所有匹配，需改为 j = next[j] 继续）
        }

        // 当前字符不匹配
        if (test[i] != pattern[j]) {
            j = next[j];   // 根据 next 数组回退 j（模式串指针）
            // 如果回退后 j 不是 -1，说明还可以继续尝试匹配，跳过 i 和 j 的自增
            if (j != -1)
                continue;  // 跳回 while 循环开头，重新比较 test[i] 和 pattern[j]
            // 若 j == -1，说明模式串已无法再左移，只能让 i 前进（见下方 ++i）
        }

        // 字符匹配成功（或 j == -1 后需要重置），主串和模式串指针都前进
        ++i;
        ++j;
    }

    // 整个主串遍历完仍未匹配成功
    cout << "匹配失败" << endl;
}
```



## 数组强化14 长度最小的子数组

给定一个含有 *n* 个正整数的数组和一个正整数 `s` ，找出该数组中满足其和≥*s* 的长度最小的 **连续** 子数组，并返回其长度。如果不存在符合条件的子数组，返回 0。

**示例**

- **输入：** `s = 7`, `nums = [2,3,1,2,4,3]`
- **输出：** `2`
- **解释：** 子数组 `[4,3]` 是该条件下的长度最小的子数组。

```c
int minSubArrayLen(int target, int* nums, int numsSize) {
    int left = 0;       // 窗口左边界
    int current_sum = 0;// 当前窗口内的和
    int min_len = INT_MAX; // 记录最小长度，初始化为最大整数值

    // right 为窗口右边界，遍历整个数组
    for (int right = 0; right < numsSize; right++) {
        // 1. 进窗口：将右边界的值加入当前和
        current_sum += nums[right];

        // 2. 判断并收缩窗口：
        // 当当前和满足条件（>= target）时，尝试移动左边界以寻找更小的长度
        while (current_sum >= target) {
            // 更新最小长度
            int window_len = right - left + 1;
            if (window_len < min_len) {
                min_len = window_len;
            }

            // 出窗口：减去左边界的值，并将左边界右移
            current_sum -= nums[left];
            left++;
        }
    }

    // 如果 min_len 仍然是初始值，说明没有找到符合条件的子数组，返回 0
    return (min_len == INT_MAX) ? 0 : min_len;
}
```



### 螺旋矩阵

给定一个正整数 n，生成一个包含 1 到 n^2 所有元素，且元素按顺时针顺序螺旋排列的正方形矩阵。

示例:

输入: 3 输出: [ [ 1, 2, 3 ], [ 8, 9, 4 ], [ 7, 6, 5 ] ]

## 

```c
#include <stdio.h>

// 定义一个足够大的全局数组，避免考试手写 malloc 的繁琐
int matrix[100][100]; 

void generate_spiral(int n) {
    int count = 1;      // 当前要填入的数字
    int start_x = 0;    // 当前圈的起始行 (top)
    int start_y = 0;    // 当前圈的起始列 (left)
    
    // 循环次数：n/2 圈
    // 如果是奇数，最后中间剩一个单独处理
    int loop = n / 2;   
    int offset = 1;     // 控制每一条边遍历的长度收缩量
    
    while (loop > 0) {
        int i, j;
        
        // 1. 上行：从左到右 (左闭右开)
        // 范围：[start_y, n - offset)
        for (j = start_y; j < n - offset; j++) {
            matrix[start_x][j] = count++;
        }
        
        // 2. 右列：从上到下 (左闭右开)
        // 范围：[start_x, n - offset)
        // 注意：此时 j 已经停在 n-offset 处（即右上角的右边），正好作为起点的列坐标逻辑延续
        for (i = start_x; i < n - offset; i++) {
            matrix[i][j] = count++;
        }
        
        // 3. 下行：从右到左 (左闭右开)
        // 范围：(start_y, n - offset] -> 倒着写就是 [n-offset, start_y)
        // 注意：此时 i 停在 n-offset 处（即右下角的下边），正好作为起点
        for (; j > start_y; j--) {
            matrix[i][j] = count++;
        }
        
        // 4. 左列：从下到上 (左闭右开)
        // 范围：(start_x, n - offset] -> 倒着写就是 [n-offset, start_x)
        // 注意：此时 j 停在 start_y 处（即左下角的左边），正好作为起点
        for (; i > start_x; i--) {
            matrix[i][j] = count++;
        }
        
        // --- 一圈结束，准备下一圈 ---
        start_x++;      // 起始点向右下移动
        start_y++;
        offset++;       // 边界收缩
        loop--;         // 圈数减 1
    }
    
    // 如果 n 是奇数，处理最中心的那个点
    if (n % 2 == 1) {
        matrix[n / 2][n / 2] = count;
    }
}

int main() {
    int n = 4; // 测试用例
    generate_spiral(n);
    
    // 打印结果
    for(int i=0; i<n; i++){
        for(int j=0; j<n; j++){
            printf("%3d ", matrix[i][j]);
        }
        printf("\n");
    }
    return 0;
}
```
